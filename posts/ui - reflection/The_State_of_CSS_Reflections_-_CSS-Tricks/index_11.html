const pg = $$('.pg') /* point groups */, 
            ig = $$('.ig') /* interv groups */, 
            gpv = $$('#gpv'), 
            gcpv = $$('.gcpv'), 
            cpv = $$('#cpv'), 
            gnp = $$('#gnp'), 
            gni = $$('#gni'), 
            cnp = $$('#cnp'), 
            cni = $$('#cni'), 
            nel = $$('#n'), m = pg.length, 
            x0 = round(1*pg[0]._attr('transform')
                        .match(/-?\d+/)[0]), 
            d = abs(2*x0);

var n, fv, pv, pvr, cpvr;

var f = function(e) {
    var prev, cond, x;
    
    if(~~nel.value !== n) {
        prev = n;
        cond = prev < n;
        n = ~~nel.value;
        fv = 1/(n - 1);
        pv = 100*fv;
        pvr = (.1*round(pv*10) + '')
                    .substring(0, 4);
        
        cnp.textContent = n;
        cni.textContent = (n - 1);
        gnp.textContent = n + ' points';
        gni.textContent = n - 1 + ' intervals';
        gpv.textContent = pvr + '%';
        cpv.textContent = pvr;
        
        for(var i = 1; i < m; i++) {
            if(i < n) {
                cpvr = (.1*round(i*pv*10) + '')
                    .substring(0, 4);
                gcpv[i].textContent = cpvr + '%';
                x = round(x0 + i*fv*d);
                pg[i]._attr({
                    'transform': 'translate(' + x + ')'
                });
                x = round(x0 + (i - .5)*fv*d);
                ig[i - 1]._attr({
                    'transform': 'translate(' + x + ')'
                });
                
                if(!cond) {
                    ig[i - 1].style.display = 
                    pg[i].style.display = 'block';
                }
            }
            else {
                if(!cond) {
                    pg[i].style.display = 
                    ig[i - 1].style.display = 'none';
                }
            }
        }
    }
};

nel.addEventListener('input', f, false);
nel.addEventListener('update', f, false);
