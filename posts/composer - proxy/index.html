Cấu hình proxy cho Composer

Thiết lập proxy cho shell.

Windows

set http_proxy=http://192.168.103.31:80
set https_proxy=http://192.168.103.31:80

echo %http_proxy%
echo %https_proxy%

set http_proxy=
set https_proxy=

Linux

export http_proxy=http://192.168.103.31:80
export https_proxy=http://192.168.103.31:80

echo $http_proxy
echo $https_proxy

unset http_proxy
unset https_proxy
