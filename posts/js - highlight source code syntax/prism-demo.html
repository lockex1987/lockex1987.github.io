Language JavaScript

const codes = [...document.querySelectorAll('pre.code')];
// Comment
codes.forEach(code => {
    const source = code.innerHTML;
    code.innerHTML = '';

    CodeMirror(code, {
        value: source,
        mode: 'javascript',
        lineNumbers: false,
        readOnly: true
    });
});
function multiply(p1, p2) {
  return p1 * p2;
}

document.getElementById('demo').innerHTML = multiply(4, 3);

Language PHP

Code PHP không tự động highlight, phải thêm prism-autoloader.

#[Attribute]
class ExampleAttribute
{
    // Comment
    public $value;
    
    public function __construct($value)
    {
        $this->value = $value;
    }
}
