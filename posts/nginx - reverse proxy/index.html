Websocket

Cấu hình tôi đã thử chạy được (sso-admin):

upstream wsbackend {
    # server 127.0.0.1:9000; # Bị lỗi
    server localhost:9000; # Chạy được
}

server {
    # Các cấu hình khác
	
    location /wsapp/ {
        proxy_pass http://wsbackend;
        # proxy_pass http://127.0.0.1:9000; # Bị lỗi
        # proxy_pass http://localhost:9000; # Chạy được

        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
		
        # keepalive_requests 500;
    }
}

Chú ý: nếu để server là localhost thì chạy được, mà để là 127.0.0.1 thì bị lỗi.

File cấu hình mẫu:

upstream plfapi {
    ip_hash;
    server 10.71.72.1:8900;
    # server 10.71.72.2:8900;
}

server {
    listen 80;
    listen 443;
    server_name localhost;
    root /data01/platform-web-frontend/public;
    index index.php index.html;

    # Để ở đây hay để ở location /api/
    # proxy_set_header Host            $host;
    # proxy_set_header X-Real-IP $remote_addr;
    # proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;

    location /gscb {
        alias /data01/gscb-web-frontend/public;

        try_files $uri $uri/ @gscb;

        location ~ \.php$ {
            include fastcgi_params;
            fastcgi_param SCRIPT_FILENAME $request_filename;
            fastcgi_pass 127.0.0.1:9000;
       }
    }

    location @gscb {
        rewrite /gscb/(.*)$ /gscb/index.php?/$1 last;
    }

    error_page 404 /index.php;

    location ~ \.php$ {
        fastcgi_pass  127.0.0.1:9000;
        fastcgi_param SCRIPT_FILENAME $realpath_root$fastcgi_script_name;
        include       fastcgi_params;
    }

    location /upload {
        root /data01/platform/pla-api/;
        secure_link $arg_md5,$arg_expires;
        secure_link_md5 "$secure_link_expires$uri pLAF@202007";
        # if ($secure_link = "") { return 403; }
        if ($secure_link = "0") { return 410; }

    }

    location /api/ {
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_pass http://plfapi;
    }
}
