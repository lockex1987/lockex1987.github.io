Chương trình đầu tiên

Soạn thảo file hello.go với nội dung sau:

package main

import "fmt"

func main() {
	fmt.Println("Hello, World")
}

Thực hiện:

go run hello.go

Build:

go build hello.go

Lệnh trên sẽ tạo file hello.exe (trên Windows) và file hello (trên Ubuntu).

Chạy file thực thi trên Windows:

hello.exe

Trên Ubuntu:

./hello

Tham số dòng lệnh

Sử dụng os.Args.

File arguments.go:

package main

import (
	"fmt"
	"os"
)

func main() {
	var s, sep string
	for i := 1; i < len(os.Args); i++ {
		s += sep + os.Args[i]
		sep = ","
	}
	fmt.Println(s)
}

Thực thi:

$ go run arguments.go ABC DEF
ABC,DEF

Biến

File variables.go:

package main

import "fmt"

func main() {
	// Khai báo, sau đó gán giá trị
	var s1 string
	s1 = "s1"

	// Vừa khai báo vừa khởi tạo
	var s2 string = "s2"

	// Có thể bỏ qua kiểu dữ liệu
	var s3 = "s3"

	// Có thể bỏ qua từ khóa var
	s4 := "s4"

	// Lần sau gán lại giá trị cho biến,
	// chúng ta chỉ sử dụng =, không sử dụng :=
	s4 = "s4 updated"

	fmt.Println(s1)
	fmt.Println(s2)
	fmt.Println(s3)
	fmt.Println(s4)
}

Hàm

Sử dụng từ khóa func.

package main

import "fmt"

func fToC(f float64) float64 {
	return (f - 32) * 5 / 9
}

func main() {
	const freezingF = 32.0
	const boilingF = 212.0
	fmt.Printf("%g°F = %g°C\n", freezingF, fToC(freezingF))
	fmt.Printf("%g°F = %g°C\n", boilingF, fToC(boilingF))
}


Gán tuple

tuple-assignment.go:

package main

import "fmt"

/**
 * Ước số chung lớn nhất (Greatest Common Divisor)
 */
func gcd(x, y int) int {
	for y != 0 {
		x, y = y, x%y
	}
	return x
}

/**
 * Fibonacci.
 */
func fib(n int) int {
	x, y := 0, 1
	for i := 0; i < n; i++ {
		x, y = y, x+y
	}
	return x
}

func main() {
	fmt.Println("Tuple")
}
